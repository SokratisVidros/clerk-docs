# Sign-up

## Basic sign-up

<div className="flex flex-col gap-10 border pt-10 rounded-xl bg-gray-25">

```tsx
<SignUp.Root>
  <SignUp.Step name="start">
    <Clerk.GlobalError />
    <Clerk.Field name="identifier">
      <Clerk.Label>Email</Clerk.Label>
      <Clerk.Input type="email" required />
      <Clerk.FieldError />
    </Clerk.Field>
    <Clerk.Field name="password">
      <Clerk.Label>Password</Clerk.Label>
      <Clerk.Input type="password" required />
      <Clerk.FieldError />
    </Clerk.Field>
    <SignUp.Action submit>Sign Up</SignUp.Action>
  </SignUp.Step>
  <SignUp.Step name="verifications">
    <Clerk.GlobalError />
    <Strategy name="phone_code">
      <Clerk.Field name="code">
        <Clerk.Label>Phone code</Clerk.Label>
        <Clerk.Input type="otp" required />
        <Clerk.FieldError />
      </Clerk.Field>
      <SignUp.Action submit>Finish registration</SignUp.Action>
    </Strategy>
  </SignUp.Step>
</SignUp.Root>
```

</div>
